// create dataabase query
CREATE DATABASE aegis;

// use database query
use aegis;

// Admin Tables
CREATE TABLE IF NOT EXISTS admins(
    id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(250) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    mobile VARCHAR(15) NOT NULL UNIQUE,
    designation VARCHAR(100) NOT NULL,
    status ENUM('active','inactive') DEFAULT 'active',
    password VARCHAR(255) NOT NULL,
    created_at VARCHAR(255),
    updated_at VARCHAR(255) DEFAULT NULL
);

// Users Tables
CREATE TABLE IF NOT EXISTS users(
    id INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(250) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    mobile VARCHAR(15) NOT NULL UNIQUE,
    alt_mobile VARCHAR(15) DEFAULT NULL,
    designation VARCHAR(100) NOT NULL,
    course VARCHAR(255),
    roll_no VARCHAR(100),
    college_name VARCHAR(555) NOT NULL,
    degree VARCHAR(255),
    semester VARCHAR(255),
    complete_year VARCHAR(100),
    address TEXT,
    city VARCHAR(100),
    state VARCHAR(255),
    pincode VARCHAR(10),
    profileImg VARCHAR(500),
    password VARCHAR(255) NOT NULL,
    status ENUM('active','inactive') DEFAULT 'active',
    document_status ENUM('active','inactive') DEFAULT 'inactive',
    created_at VARCHAR(255),
    updated_at VARCHAR(255) DEFAULT NULL
);
